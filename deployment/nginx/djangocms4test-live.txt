
# redirects ftw
server {
    listen 127.0.0.1:62321;
    server_name
        djangocms4test.ch
        # djangocms4test.live.bnzk.ch
        ;
    rewrite ^/(.*) http://djangocms4test.ch/$1 permanent;
}

upstream djangocms4test_live_server {
    server unix:/home/djangocms4test/sites/djangocms4test-live.sock;
}

server {
    listen 127.0.0.1:62321;
    server_name
        djangocms4test.ch
        djangocms4test.live.bnzk.ch
        ;

    location / {
        proxy_set_header Host $http_host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        # x-forwarded-proto for gunicorn, x-forwarded-protocol for Django!
        # $scheme is http, as the front nginx is doing ssl termination!
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_redirect off;
        if (!-f $request_filename) {
            proxy_pass      http://djangocms4test_live_server ;
            break;
        }
    }

    error_log logs/djangocms4test-live-error.log;
    access_log logs/djangocms4test-live-access.log;

    error_page 500 502 503 504 /nginx/50x.html;
    location /nginx/ {
        root    /home/djangocms4test/sites/djangocms4test-live/public/static/;
    }
    location /static/ {
        alias   /home/djangocms4test/sites/djangocms4test-live/public/static/;
        expires 3M;
    }
    location /media/ {
        alias   /home/djangocms4test/sites/djangocms4test-live/public/media/;
        expires 3M;
    }
}
